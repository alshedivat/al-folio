8/**
9 * @brief Sets the initial values for the INTERIOR of a specified Eulerian field.
10 *
11 * This function initializes the interior nodes of `Ucont` based on a profile selected
12 * by `user->FieldInitialization`. It explicitly skips any node that lies on a global
13 * boundary, as those values are set by the Boundary System's `Initialize` methods.
14 *
15 * The initialization is directional, aligned with the primary INLET face that was
16 * identified by the parser. This ensures the initial flow is physically meaningful.
17 *
18 * Supported `user->FieldInitialization` profiles for "Ucont":
19 * - 0: Zero Velocity. All interior components of Ucont are set to 0.
20 * - 1: Constant Normal Velocity. The contravariant velocity component normal to the
21 * inlet direction is set such that the physical velocity normal to those grid
22 * planes is a constant `uin`. Other contravariant components are zero.
23 * - 2: Poiseuille Normal Velocity. The contravariant component normal to the
24 * inlet direction is set with a parabolic profile.
25 *
26 * @param user The main UserCtx struct, containing all simulation data and configuration.
27 * @param fieldName A string ("Ucont" or "P") identifying which field to initialize.

239 * @brief (HELPER) Finalizes a block's state by applying BCs, converting to Cartesian,
240 * and synchronizing ghost regions.
241 */
242static PetscErrorCode FinalizeBlockState(UserCtx *user)
243{
244 PetscErrorCode ierr;
245 PetscFunctionBeginUser;
246
247 // This sequence ensures a fully consistent state for a single block.
248 // 1. Apply BCs using the information from InflowFlux/OutflowFlux.
249 ierr = FormBCS(user); CHKERRQ(ierr);
250 LOG_ALLOW(GLOBAL,LOG_DEBUG," Boundary condition applied.\n");
251 // 2. Sync contravariant velocity field.
252 ierr = UpdateLocalGhosts(user, "Ucont"); CHKERRQ(ierr);
253 LOG_ALLOW(GLOBAL,LOG_DEBUG," Ucont field ghosts updated.\n");
254
255 // 3. Convert to Cartesian velocity.
256 ierr = Contra2Cart(user); CHKERRQ(ierr);
257 LOG_ALLOW(GLOBAL,LOG_DEBUG," Converted Ucont to Ucat.\n");
258
259 // 4. Sync the new Cartesian velocity field.
260 ierr = UpdateLocalGhosts(user, "Ucat"); CHKERRQ(ierr);
261 LOG_ALLOW(GLOBAL,LOG_DEBUG," Ucat field ghosts updated.\n");
262
263 PetscFunctionReturn(0);
264}

267/**
268 * @brief (HELPER) Sets the t=0 fluid state for all blocks.
269 *
270 * This function replicates the initialization sequence for a fresh start from
271 * the legacy code. It sets an initial guess for the interior, establishes the
272 * boundary conditions, and ensures the final state is consistent across MPI ranks.
273 */
274static PetscErrorCode SetInitialFluidState_FreshStart(SimCtx *simCtx)
275{
276 PetscErrorCode ierr;
277 UserCtx *user_finest = simCtx->usermg.mgctx[simCtx->usermg.mglevels - 1].user;
278 PetscFunctionBeginUser;
279
280 for (PetscInt bi = 0; bi < simCtx->block_number; bi++) {
281 LOG_ALLOW(LOCAL, LOG_INFO, "Rank %d, Block %d: Setting t=0 state.\n", simCtx->rank, bi);
282
283 // 1. Set an initial guess for the INTERIOR of the domain.
284 // Replaces the legacy `if(InitialGuessOne)` block.
285
286 LOG_ALLOW(GLOBAL,LOG_DEBUG," Initializing Interior Ucont field.\n");
287 ierr = SetInitialInteriorField(&user_finest[bi], "Ucont"); CHKERRQ(ierr);
288 LOG_ALLOW(GLOBAL,LOG_DEBUG," Interior Ucont field initialized.\n");
289
290 // 2. Establish the initial boundary flux values and set inlet profiles.
291 // This sequence is critical and comes directly from the legacy setup.
292 LOG_ALLOW(GLOBAL,LOG_DEBUG," Boundary condition Preparation steps initiated.\n");
293 ierr = InflowFlux(&user_finest[bi]); CHKERRQ(ierr);
294 ierr = OutflowFlux(&user_finest[bi]); CHKERRQ(ierr);
295 LOG_ALLOW(GLOBAL,LOG_DEBUG," Boundary condition Preparation steps completed.\n");
296 // 3. Apply all boundary conditions, convert to Cartesian, and sync ghosts.
297 LOG_ALLOW(GLOBAL,LOG_DEBUG," Boundary condition application and state finalization initiated.\n");
298 ierr = FinalizeBlockState(&user_finest[bi]); CHKERRQ(ierr);
299 LOG_ALLOW(GLOBAL,LOG_DEBUG," Boundary condition application and state finalization complete.\n");
300 }
301
302 // If using multiple grid blocks, handle the interface conditions between them.
303 if (simCtx->block_number > 1) {
304 // LOG_ALLOW(GLOBAL, LOG_INFO, "Updating multi-block interfaces for t=0.\n");
305 // ierr = Block_Interface_U(user_finest); CHKERRQ(ierr);
306 // After interface update, ghost regions might be stale. Refresh them.
307 for (PetscInt bi = 0; bi < simCtx->block_number; bi++) {
308 ierr = UpdateLocalGhosts(&user_finest[bi], "Ucont"); CHKERRQ(ierr);
309 ierr = UpdateLocalGhosts(&user_finest[bi], "Ucat"); CHKERRQ(ierr);
310 }
311 }
312
313 PetscFunctionReturn(0);
314}

343/**
344 * @brief High-level orchestrator to set the complete initial state of the Eulerian solver.
345 *
346 * This function is called once from main() before the time loop begins. It inspects
347 * the simulation context to determine whether to perform a fresh start (t=0) or
348 * restart from saved files. It then delegates to the appropriate helper function.
349 * Finally, it initializes the solver's history vectors (Ucont_o, P_o, etc.)
350 * to ensure the first time step has the necessary data.
351 */
352PetscErrorCode InitializeEulerianState(SimCtx *simCtx)
353{
354 PetscErrorCode ierr;
355 UserCtx *user_finest = simCtx->usermg.mgctx[simCtx->usermg.mglevels - 1].user;
356
357 PetscFunctionBeginUser;
358
359 LOG_ALLOW(GLOBAL, LOG_INFO, "--- Initializing Eulerian State ---\n");
360
361 if (simCtx->rstart_flg) {
362 LOG_ALLOW(GLOBAL, LOG_INFO, "Starting from RESTART files (t=%.4f, step=%d).\n",
363 simCtx->StartTime, simCtx->StartStep);
364 ierr = SetInitialFluidState_Restart(simCtx); CHKERRQ(ierr);
365 } else {
366 LOG_ALLOW(GLOBAL, LOG_INFO, "Performing a FRESH START (t=0, step=0).\n");
367 ierr = SetInitialFluidState_FreshStart(simCtx); CHKERRQ(ierr);
368 }
369
370 // This crucial step, taken from the end of the legacy setup, ensures
371 // that the history vectors (Ucont_o, Ucont_rm1, etc.) are correctly
372 // populated before the first call to the time-stepping loop.
373 for (PetscInt bi = 0; bi < simCtx->block_number; bi++) {
374 ierr = UpdateSolverHistoryVectors(&user_finest[bi]); CHKERRQ(ierr);
375 }
376
377 LOG_ALLOW(GLOBAL, LOG_INFO, "--- Eulerian State Initialized and History Vectors Populated ---\n");
378 PetscFunctionReturn(0);
379}
