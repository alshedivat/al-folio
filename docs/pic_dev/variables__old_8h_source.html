114/**
115 * @enum ParticleLocationStatus
116 * @brief Defines the state of a particle with respect to its location and migration
117 * status during the iterative SettleParticles() process.
118 *
119 * This enum is used to control the logic within the main particle relocation
120 * orchestrator, ensuring that each particle is processed correctly and efficiently.
121 * It is registered as a PETSC_INT field in the DMSwarm.
122 */
123typedef enum {
124 /**
125 * @brief The particle's host cell is unknown or invalid.
126 * This is the initial state for newly arrived particles on a rank after migration,
127 * or for particles whose location search failed in a previous step. Any particle
128 * in this state is a candidate for processing in the current relocation pass.
129 */
130 NEEDS_LOCATION,
131
132 /**
133 * @brief The particle has been successfully located in a cell owned by the current rank.
134 * This state indicates that the particle is settled for the current timestep and
135 * does not need further location checks or migration. The orchestrator can skip
136 * this particle in subsequent passes of the do-while loop.
137 */
138 ACTIVE_AND_LOCATED,
139
140 /**
141 * @brief The particle has been identified as needing to move to another rank.
142 * This status is set by the sending rank during the identification phase. The particle's
143 * `destination_rank` field will be valid. This state is temporary and primarily used
144 * to signal that the particle should be included in the MPI migration step.
145 */
146 MIGRATING_OUT,
147
148 /**
149 * @brief The particle could not be located and is considered lost.
150 * This status is set when the walk search exceeds its maximum traversal limit or
151 * walks outside the global domain boundaries. Particles in this state are candidates
152 * for removal by the `CheckAndRemoveOutOfBoundsParticles` function.
153 */
154 LOST,
155
156 /**
157 * @brief The default state for a newly created particle before any processing.
158 * While `NEEDS_LOCATION` is functionally similar for the first pass, having a distinct
159 * initial state can be useful for debugging and initialization logic.
160 */
161 UNINITIALIZED
162
163} ParticleLocationStatus;
