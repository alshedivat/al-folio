504/**
505 * @brief Removes particles that have been definitively flagged as LOST by the location algorithm.
506 *
507 * This function is the designated cleanup utility for particles that have exited the
508 * physical domain or could not be located for any other reason. It should be called
509 * after a particle location and migration phase is complete.
510 *
511 * It iterates through all locally owned particles and checks their `DMSwarm_location_status`
512 * field. If a particle's status is `LOST`, it is permanently removed from the simulation
513 * using `DMSwarmRemovePointAtIndex`.
514 *
515 * The function is carefully designed to handle modifications to the DMSwarm during iteration
516 * safely. It does this by:
517 * 1. Iterating BACKWARDS through the local particle list, so removing an element at
518 * index 'p' does not affect the indices of subsequent elements to be checked (p-1, p-2, ...).
519 * 2. Using a robust "Restore-Remove-Reacquire" pattern. When a particle is removed,
520 * all pointers to swarm data are first restored, the removal operation is performed,
521 * and then the pointers are re-acquired for the remainder of the loop.
522 *
523 * @warning This function contains a collective MPI operation (MPI_Allreduce) at the end.
524 * To avoid deadlocks, it MUST be called by ALL ranks in the communicator, even if
525 * a rank has no local particles. Do not place this call inside rank-specific
526 * conditional logic in your main loop.
527 *
528 * @param[in,out] user Pointer to the UserCtx structure containing the swarm.
529 * @param[out] removedCountLocal Pointer to an integer that will store the number of particles removed on THIS rank.
530 * @param[out] removedCountGlobal Pointer to an integer that will store the total number of particles removed ACROSS ALL ranks.
531 *
532 * @return PetscErrorCode 0 on success, or a non-zero PETSc error code on failure.
533 */
534PetscErrorCode CheckAndRemoveLostParticles(UserCtx *user,
535 PetscInt *removedCountLocal,
536 PetscInt *removedCountGlobal)
537{
538 PetscErrorCode ierr;
539 DM swarm = user->swarm;
540 PetscInt nLocalInitial;
541 PetscInt *status_p = NULL;
542 PetscInt64 *pid_p = NULL; // For better logging
543 PetscReal *pos_p = NULL; // For better logging
544 PetscInt local_removed_count = 0;
545 PetscMPIInt global_removed_count_mpi = 0;
546 PetscMPIInt rank;
547
548 PetscFunctionBeginUser;
549 ierr = MPI_Comm_rank(PETSC_COMM_WORLD, &rank); CHKERRQ(ierr);
550 LOG_ALLOW(LOCAL, LOG_INFO, "Rank %d: Checking for and removing LOST particles...", rank);
551
552 // Initialize output parameters to ensure clean state
553 *removedCountLocal = 0;
554 if (removedCountGlobal) *removedCountGlobal = 0;
555
556 ierr = DMSwarmGetLocalSize(swarm, &nLocalInitial); CHKERRQ(ierr);
557
558 // Only proceed if there are particles to check on this rank.
559 // All ranks will still participate in the final collective MPI_Allreduce.
560 if (nLocalInitial > 0) {
561 // Get access to all swarm fields once before the loop begins.
562 ierr = DMSwarmGetField(swarm, "DMSwarm_location_status", NULL, NULL, (void **)&status_p); CHKERRQ(ierr);
563 ierr = DMSwarmGetField(swarm, "DMSwarm_pid", NULL, NULL, (void **)&pid_p); CHKERRQ(ierr);
564 ierr = DMSwarmGetField(swarm, "position", NULL, NULL, (void **)&pos_p); CHKERRQ(ierr);
565
566 // --- Iterate BACKWARDS to handle index changes safely during removal ---
567 for (PetscInt p = nLocalInitial - 1; p >= 0; p--) {
568 if (status_p[p] == LOST) {
569 LOG_ALLOW(LOCAL, LOG_DEBUG, "Rank %d: Removing LOST particle [PID %lld] at local index %d. Position: (%.4f, %.4f, %.4f).\n",
570 rank, (long long)pid_p[p], p, pos_p[3*p], pos_p[3*p+1], pos_p[3*p+2]);
571
572 // --- Safe Removal Pattern: Restore -> Remove -> Reacquire ---
573 // This is the fix for the double-restore bug. Pointers are managed carefully
574 // within this block and then restored cleanly after the loop.
575
576 // 1. Restore all fields BEFORE modifying the swarm structure. This invalidates all pointers.
577 ierr = DMSwarmRestoreField(swarm, "DMSwarm_location_status", NULL, NULL, (void **)&status_p); CHKERRQ(ierr);
578 ierr = DMSwarmRestoreField(swarm, "DMSwarm_pid", NULL, NULL, (void **)&pid_p); CHKERRQ(ierr);
579 ierr = DMSwarmRestoreField(swarm, "position", NULL, NULL, (void **)&pos_p); CHKERRQ(ierr);
580
581 // 2. Remove the particle at the current local index 'p'.
582 ierr = DMSwarmRemovePointAtIndex(swarm, p); CHKERRQ(ierr);
583 local_removed_count++;
584
585 // 3. After removal, re-acquire pointers ONLY if the loop is not finished.
586 PetscInt nLocalCurrent;
587 ierr = DMSwarmGetLocalSize(swarm, &nLocalCurrent); CHKERRQ(ierr);
588
589 if (nLocalCurrent > 0 && p > 0) { // Check if there are particles left AND iterations left
590 ierr = DMSwarmGetField(swarm, "DMSwarm_location_status", NULL, NULL, (void **)&status_p); CHKERRQ(ierr);
591 ierr = DMSwarmGetField(swarm, "DMSwarm_pid", NULL, NULL, (void **)&pid_p); CHKERRQ(ierr);
592 ierr = DMSwarmGetField(swarm, "position", NULL, NULL, (void **)&pos_p); CHKERRQ(ierr);
593 } else {
594 // All remaining particles were removed OR this was the last particle (p=0).
595 // Invalidate pointers to prevent the final restore call and exit the loop.
596 status_p = NULL;
597 pid_p = NULL;
598 pos_p = NULL;
599 break;
600 }
601 }
602 } // End of backwards loop
603
604 // At the end, restore any valid pointers. This handles three cases:
605 // 1. No particles were removed: restores the original pointers.
606 // 2. Particles were removed mid-loop: restores the pointers from the last re-acquisition.
607 // 3. All particles were removed: pointers are NULL, so nothing is done.
608 if (status_p) { ierr = DMSwarmRestoreField(swarm, "DMSwarm_location_status", NULL, NULL, (void **)&status_p); CHKERRQ(ierr); }
609 if (pid_p) { ierr = DMSwarmRestoreField(swarm, "DMSwarm_pid", NULL, NULL, (void **)&pid_p); CHKERRQ(ierr); }
610 if (pos_p) { ierr = DMSwarmRestoreField(swarm, "position", NULL, NULL, (void **)&pos_p); CHKERRQ(ierr); }
611 } // End of if (nLocalInitial > 0)
612
613 PetscInt nLocalFinal;
614 ierr = DMSwarmGetLocalSize(swarm, &nLocalFinal); CHKERRQ(ierr);
615 LOG_ALLOW(LOCAL, LOG_INFO, "Rank %d: Finished removing %d LOST particles. Final local size: %d.\n", rank, local_removed_count, nLocalFinal);
616
617 // --- Synchronize counts across all ranks ---
618 *removedCountLocal = local_removed_count;
619 if (removedCountGlobal) {
620 ierr = MPI_Allreduce(&local_removed_count, &global_removed_count_mpi, 1, MPI_INT, MPI_SUM, PetscObjectComm((PetscObject)swarm)); CHKERRQ(ierr);
621 *removedCountGlobal = global_removed_count_mpi;
622 // Use a synchronized log message so only one rank prints the global total.
623 LOG_ALLOW_SYNC(GLOBAL, LOG_INFO, "[Rank %d] Removed %d LOST particles globally.\n", rank, *removedCountGlobal);
624 }
625
626 PetscFunctionReturn(0);
627}

1074// --- Helper function to resize swarm globally (add or remove) ---
1075// This assumes removing excess particles means removing the globally last ones.
1076PetscErrorCode ResizeSwarmGlobally(DM swarm, PetscInt N_target)
1077{
1078 PetscErrorCode ierr;
1079 PetscInt N_current, nlocal_current;
1080 PetscMPIInt rank;
1081 MPI_Comm comm;
1082
1083 PetscFunctionBeginUser;
1084 ierr = PetscObjectGetComm((PetscObject)swarm, &comm); CHKERRQ(ierr);
1085 ierr = MPI_Comm_rank(comm, &rank); CHKERRQ(ierr);
1086 ierr = DMSwarmGetSize(swarm, &N_current); CHKERRQ(ierr);
1087 ierr = DMSwarmGetLocalSize(swarm, &nlocal_current); CHKERRQ(ierr);
1088
1089 PetscInt delta = N_target - N_current;
1090
1091 if (delta == 0) {
1092 PetscFunctionReturn(0); // Nothing to do
1093 }
1094
1095 if (delta < 0) { // Remove particles
1096 PetscInt num_to_remove_global = -delta;
1097 LOG_ALLOW(GLOBAL, LOG_INFO, "ResizeSwarmGlobally: Current size %d > target size %d. Removing %d particles globally.\n", N_current, N_target, num_to_remove_global);
1098
1099 // --- Strategy: Remove the globally last 'num_to_remove_global' particles ---
1100 // Each rank needs to determine how many of its *local* particles fall
1101 // within the range of global indices [N_target, N_current - 1].
1102
1103 PetscInt rstart = 0;
1104 PetscInt rend;
1105 // Global range owned by this rank [rstart, rend)
1106
1107 ierr = MPI_Exscan(&nlocal_current, &rstart, 1, MPIU_INT, MPI_SUM, comm); CHKERRMPI(ierr); // Use CHKERRMPI for MPI calls
1108
1109 rend = rstart + nlocal_current;
1110
1111
1112 // Calculate how many local particles have global indices >= N_target
1113 PetscInt nlocal_remove_count = 0;
1114 if (rend > N_target) { // If this rank owns any particles slated for removal
1115 PetscInt start_remove_local_idx = (N_target > rstart) ? (N_target - rstart) : 0;
1116 nlocal_remove_count = nlocal_current - start_remove_local_idx;
1117 }
1118
1119 if (nlocal_remove_count < 0) nlocal_remove_count = 0; // Sanity check
1120 if (nlocal_remove_count > nlocal_current) nlocal_remove_count = nlocal_current; // Sanity check
1121
1122 LOG_ALLOW(LOCAL, LOG_DEBUG, "Rank %d: Global range [%d, %d). Target size %d. Need to remove %d local particles (from end).\n", rank, rstart, rend, N_target, nlocal_remove_count);
1123
1124 // Remove the last 'nlocal_remove_count' particles *locally* by iterating backwards
1125 PetscInt removal_ops_done = 0;
1126 for (PetscInt p = nlocal_current - 1; p >= 0 && removal_ops_done < nlocal_remove_count; --p) {
1127 ierr = DMSwarmRemovePointAtIndex(swarm, p); CHKERRQ(ierr);
1128 removal_ops_done++;
1129 }
1130
1131 if (removal_ops_done != nlocal_remove_count) {
1132 SETERRQ(comm, PETSC_ERR_PLIB, "Rank %d: Failed to remove the expected number of local particles (%d != %d)", rank, removal_ops_done, nlocal_remove_count);
1133 }
1134 LOG_ALLOW(LOCAL, LOG_DEBUG, "Rank %d: Removed %d local particles.\n", rank, removal_ops_done);
1135
1136 // Barrier to ensure all removals are done before size check
1137 ierr = MPI_Barrier(comm); CHKERRMPI(ierr);
1138
1139 } else { // delta > 0: Add particles
1140 PetscInt num_to_add_global = delta;
1141 LOG_ALLOW(GLOBAL, LOG_INFO, "ResizeSwarmGlobally: Current size %d < target size %d. Adding %d particles globally.\n", N_current, N_target, num_to_add_global);
1142 ierr = DMSwarmAddNPoints(swarm, num_to_add_global); CHKERRQ(ierr);
1143 // Note: Added particles will have uninitialized field data. Reading will overwrite.
1144 }
1145
1146 // Verify final size
1147 PetscInt N_final;
1148 ierr = DMSwarmGetSize(swarm, &N_final); CHKERRQ(ierr);
1149 if (N_final != N_target) {
1150 SETERRQ(comm, PETSC_ERR_PLIB, "Failed to resize swarm: expected %d particles, got %d", N_target, N_final);
1151 }
1152 LOG_ALLOW(GLOBAL, LOG_DEBUG, "ResizeSwarmGlobally: Swarm successfully resized to %d particles.\n", N_final);
1153 PetscFunctionReturn(0);
1154}

1549 * @return PetscErrorCode 0 on success, or a non-zero PETSc error code on failure.
1550 */

1552 PetscInt n_local,
1553 PetscInt64 **pids_snapshot_out)

1556 PetscMPIInt rank;
1557
1558 PetscFunctionBeginUser;

1595/**
1596 * @brief Safely adds a new migration task to a dynamically sized list.
1597 *
1598 * This utility function manages a dynamic array of MigrationInfo structs. It appends
1599 * a new entry to the list and automatically doubles the array's capacity using
1600 * `PetscRealloc` if the current capacity is exceeded. This prevents buffer overflows
1601 * and avoids the need to know the number of migrating particles in advance.
1602 *
1603 * @param[in,out] migration_list_p A pointer to the MigrationInfo array pointer. The function
1604 * will update this pointer if the array is reallocated.
1605 * @param[in,out] capacity_p A pointer to an integer holding the current allocated
1606 * capacity of the list (in number of elements). This will be
1607 * updated upon reallocation.
1608 * @param[in,out] count_p A pointer to an integer holding the current number of
1609 * items in the list. This will be incremented by one.
1610 * @param[in] particle_local_idx The local index (from 0 to nlocal-1) of the particle
1611 * that needs to be migrated.
1612 * @param[in] destination_rank The target MPI rank for the particle.
1613 *
1614 * @return PetscErrorCode 0 on success, or a non-zero PETSc error code on failure (e.g., from memory allocation).
1615 */
1616PetscErrorCode AddToMigrationList(MigrationInfo **migration_list_p,
1617 PetscInt *capacity_p,
1618 PetscInt *count_p,
1619 PetscInt particle_local_idx,
1620 PetscMPIInt destination_rank)
1621{
1622 PetscErrorCode ierr;
1623 PetscMPIInt rank;
1624
1625 PetscFunctionBeginUser;
1626
1627 // --- 1. Input Validation ---
1628 if (!migration_list_p || !capacity_p || !count_p) {
1629 SETERRQ(PETSC_COMM_SELF, PETSC_ERR_ARG_NULL, "Null pointer provided to AddToMigrationList for list management.");
1630 }
1631
1632 // --- 2. Check if the list needs to be resized ---
1633 if (*count_p >= *capacity_p) {
1634 PetscInt old_capacity = *capacity_p;
1635 // Start with a reasonable base capacity, then double for subsequent reallocations.
1636 PetscInt new_capacity = (old_capacity == 0) ? 16 : old_capacity * 2;
1637
1638 // Use PetscRealloc for safe memory reallocation.
1639 // It handles allocating new memory, copying old data, and freeing the old block.
1640 // The first argument to PetscRealloc is the new size in BYTES.
1641 ierr = PetscRealloc(new_capacity * sizeof(MigrationInfo), migration_list_p); CHKERRQ(ierr);
1642
1643 *capacity_p = new_capacity; // Update the capacity tracker
1644
1645 ierr = MPI_Comm_rank(PETSC_COMM_WORLD, &rank); CHKERRQ(ierr);
1646 LOG_ALLOW(LOCAL, LOG_DEBUG, "AddToMigrationList [Rank %d]: Reallocated migrationList capacity from %d to %d.\n",
1647 rank, old_capacity, new_capacity);
1648 }
1649
1650 // --- 3. Add the new migration data to the list ---
1651 // Dereference the pointer-to-a-pointer to get the actual array.
1652 MigrationInfo *list = *migration_list_p;
1653
1654 list[*count_p].local_index = particle_local_idx;
1655 list[*count_p].target_rank = destination_rank;
1656
1657 // --- 4. Increment the count of items in the list ---
1658 (*count_p)++;
1659
1660 PetscFunctionReturn(0);
1661}

1663/**
1664 * @brief Identifies newly arrived particles after migration and flags them for a location search.
1665 * @ingroup ParticleMotion
1666 *
1667 * This function is a critical component of the iterative migration process managed by
1668 * the main particle settlement orchestrator (e.g., `SettleParticles`). After a
1669 * `DMSwarmMigrate` call, each rank's local particle list is a new mix of resident
1670 * particles and newly received ones. This function's job is to efficiently identify
1671 * these "newcomers" and set their `DMSwarm_location_status` field to `NEEDS_LOCATION`.
1672 *
1673 * This ensures that in the subsequent pass of the migration `do-while` loop, only the
1674 * newly arrived particles are processed by the expensive location algorithm, preventing
1675 * redundant work on particles that are already settled on the current rank.
1676 *
1677 * The identification is done by comparing the PIDs of particles currently on the rank
1678 * against a "snapshot" of PIDs taken *before* the migration occurred.
1679 *
1680 * @param[in] swarm The DMSwarm object, which has just completed a migration.
1681 * @param[in] n_local_before The number of particles that were on this rank *before* the
1682 * migration was performed.
1683 * @param[in] pids_before A pre-sorted array of the PIDs that were on this rank before
1684 * the migration. This is used for fast lookups.
1685 *
1686 * @return PetscErrorCode 0 on success, or a non-zero PETSc error code on failure.
1687 *
1688 * @note This function assumes the `pids_before` array is sorted in ascending order to
1689 * enable the use of an efficient binary search.
1690 */
1691PetscErrorCode FlagNewcomersForLocation(DM swarm,
1692 PetscInt n_local_before,
1693 const PetscInt64 pids_before[])
1694{
1695 PetscErrorCode ierr;
1696 PetscMPIInt rank;
1697 PetscInt n_local_after;
1698 PetscInt newcomer_count = 0;
1699
1700 // Pointers to the swarm data fields we will read and modify
1701 PetscInt64 *pid_field_after = NULL;
1702 PetscInt *status_field_after = NULL;
1703 PetscInt *cell_field_after = NULL;
1704
1705 PetscFunctionBeginUser;
1706
1707 // --- 1. Input Validation and Basic Setup ---
1708 if (!swarm) {
1709 SETERRQ(PETSC_COMM_SELF, PETSC_ERR_ARG_WRONGSTATE, "Input DMSwarm is NULL in FlagNewcomersForLocation.");
1710 }
1711 // If n_local_before > 0, the corresponding PID array must not be null.
1712 if (n_local_before > 0 && !pids_before) {
1713 SETERRQ(PETSC_COMM_SELF, PETSC_ERR_ARG_NULL, "Input pids_before array is NULL for n_local_before > 0.");
1714 }
1715
1716 ierr = MPI_Comm_rank(PETSC_COMM_WORLD, &rank); CHKERRQ(ierr);
1717
1718 // Get the number of particles on this rank *after* the migration.
1719 ierr = DMSwarmGetLocalSize(swarm, &n_local_after); CHKERRQ(ierr);
1720
1721 LOG_ALLOW(LOCAL, LOG_DEBUG, "FlagNewcomersForLocation [Rank %d]: Checking for newcomers. Size before: %d, Size after: %d\n",
1722 rank, n_local_before, n_local_after);
1723
1724 // If there are no particles now, there's nothing to do.
1725 if (n_local_after == 0) {
1726 PetscFunctionReturn(0);
1727 }
1728
1729 // --- 2. Access Swarm Data ---
1730 // Get read-only access to the PIDs and read-write access to the status field.
1731 ierr = DMSwarmGetField(swarm, "DMSwarm_pid", NULL, NULL, (void**)&pid_field_after); CHKERRQ(ierr);
1732 ierr = DMSwarmGetField(swarm, "DMSwarm_location_status", NULL, NULL, (void**)&status_field_after); CHKERRQ(ierr);
1733 ierr = DMSwarmGetField(swarm, "DMSwarm_CellID", NULL, NULL, (void**)&cell_field_after); CHKERRQ(ierr);
1734 if (!pid_field_after || !status_field_after) {
1735 SETERRQ(PETSC_COMM_SELF, PETSC_ERR_ARG_WRONGSTATE, "Failed to get required swarm fields in FlagNewcomersForLocation.");
1736 }
1737
1738 // --- 3. Identify and Flag Newcomers ---
1739 // Loop through all particles currently on this rank.
1740 for (PetscInt p_idx = 0; p_idx < n_local_after; ++p_idx) {
1741 PetscInt64 current_pid = pid_field_after[p_idx];
1742 PetscBool is_found_in_before_list;
1743
1744 // Use our custom, efficient helper function for the lookup.
1745 ierr = BinarySearchInt64(n_local_before, pids_before, current_pid, &is_found_in_before_list); CHKERRQ(ierr);
1746
1747 // If the PID was NOT found in the "before" list, it must be a newcomer.
1748 if (!is_found_in_before_list) {
1749 // Flag it for processing in the next pass of the migration loop.
1750 status_field_after[p_idx] = NEEDS_LOCATION;
1751 // cell_field_after[3*p_idx+0] = -1;
1752 // cell_field_after[3*p_idx+1] = -1;
1753 // cell_field_after[3*p_idx+2] = -1;
1754 newcomer_count++;
1755
1756 LOG_ALLOW(LOCAL, LOG_DEBUG, "[Rank %d]: Flagged newcomer PID %ld at local index %d as NEEDS_LOCATION.\n",
1757 rank, current_pid, p_idx);
1758 }
1759 }
1760
1761 // --- 4. Restore Swarm Fields ---
1762 // Release the locks on the swarm data arrays.
1763 ierr = DMSwarmRestoreField(swarm, "DMSwarm_pid", NULL, NULL, (void**)&pid_field_after); CHKERRQ(ierr);
1764 ierr = DMSwarmRestoreField(swarm, "DMSwarm_location_status", NULL, NULL, (void**)&status_field_after); CHKERRQ(ierr);
1765 ierr = DMSwarmRestoreField(swarm, "DMSwarm_CellID", NULL, NULL, (void**)&cell_field_after); CHKERRQ(ierr);
1766
1767 if (newcomer_count > 0) {
1768 LOG_ALLOW(LOCAL, LOG_INFO, "[Rank %d]: Identified and flagged %d newcomers.\n", rank, newcomer_count);
1769 }
1770
1771 PetscFunctionReturn(0);
1772}

1776 * @ingroup ParticleLocation
1777 *
1778 * This function is part of the "Guess and Verify" strategy, called only for "lost"
1779 * particles. It attempts to find a candidate owner by checking which rank's bounding box
1780 * contains the particle's physical position.
1781 *
1782 * To optimize the search, it uses the particle's position relative to the local
1783 * bounding box to intelligently check the most likely neighboring ranks first.
1784 * For example, if a particle's x-coordinate is less than the local minimum x, it
1785 * will check the -X neighbor first. If no owner is found in the immediate neighbors,
1786 * it performs a full search of all other ranks as a fallback.
1787 *
1788 * @param[in] user Pointer to the UserCtx, which must contain the pre-computed
1789 * `bbox` (local), `neighbors` struct, and the global `bboxlist`.
1790 * @param[in] particle A pointer to the particle whose owner is being guessed.
1791 * @param[in] bboxlist An array of BoundingBox structures for ALL MPI ranks, indexed 0 to (size-1).
1792 * This array must be up-to-date and available on all ranks.
1793 * @param[out] guess_rank_out A pointer to a PetscMPIInt. Set to the candidate owner's rank
1794 * if found, otherwise set to -1 (or MPI_PROC_NULL).
1795 *
1796 * @return PetscErrorCode 0 on success, or a non-zero PETSc error code on failure.
1797 */
1798PetscErrorCode GuessParticleOwnerWithBBox(UserCtx *user,
1799 const Particle *particle,
1800 const BoundingBox *bboxlist,
1801 PetscMPIInt *guess_rank_out)
1802{
1803 PetscErrorCode ierr;
1804 PetscMPIInt rank, size;
1805 const RankNeighbors *neighbors = &user->neighbors; // Use a direct pointer for clarity
1806 const BoundingBox *localBBox = &user->bbox;
1807
1808 PetscFunctionBeginUser;
1809
1810 // --- 1. Input Validation and Setup ---
1811 if (!user || !particle || !guess_rank_out || !bboxlist) {
1812 SETERRQ(PETSC_COMM_SELF, PETSC_ERR_ARG_NULL, "Null pointer provided to GuessParticleOwnerWithBBox.");
1813 }
1814 if (!localBBox|| !neighbors) {
1815 SETERRQ(PETSC_COMM_SELF, PETSC_ERR_ARG_WRONGSTATE, "Required user->bboxl or user->neighbors is not initialized.");
1816 }
1817
1818 ierr = MPI_Comm_rank(PETSC_COMM_WORLD, &rank); CHKERRQ(ierr);
1819 ierr = MPI_Comm_size(PETSC_COMM_WORLD, &size); CHKERRQ(ierr);
1820
1821 *guess_rank_out = MPI_PROC_NULL; // Default to "not found"
