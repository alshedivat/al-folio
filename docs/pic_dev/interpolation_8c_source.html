246/**
247 * @brief Interpolates a scalar field from cell centers to corner nodes.
248 *
249 * This function estimates the value of a scalar field at each grid node by averaging

255 * Input array uses GLOBAL cell indices, output array uses GLOBAL node indices.
256 *
257 * @param[in] centfield_arr Input: 3D array (ghosted) of scalar data at cell centers,

260 * accessed via GLOBAL node indices (k=0..KM, j=0..JM, i=0..IM).

262 *
263 * @return PetscErrorCode 0 on success.
264 */
265PetscErrorCode InterpolateFieldFromCenterToCorner_Scalar(
266 PetscReal ***centfield_arr, /* Input: Ghosted local array based on da (cell indices) */
267 PetscReal ***field_arr, /* Output: Ghosted local array based on da (node indices) */
268 UserCtx *user)
269{
270 PetscErrorCode ierr;
271 PetscMPIInt rank;

273 LOG_ALLOW_SYNC(GLOBAL, LOG_DEBUG,
274 "Rank %d starting interpolation.\n", rank);

276 // Get local info based on the DMDA (da). This info primarily describes owned nodes.

284 PetscInt ys_node = info.ys;

297 LOG_ALLOW(GLOBAL, LOG_DEBUG, "Rank %d: Interpolating for owned NODES k=%d..%d, j=%d..%d, i=%d..%d\n",

299
300 // Loop over the GLOBAL indices of the NODES owned by this processor
301 for (PetscInt k = zs_node; k < ze_node; k++) { // Global NODE index k (0..KM)

321 ck >= 0 && ck <= KM-1)
322 {

324 // Relies on centfield_arr being from a ghosted local vector.
325 sum += centfield_arr[ck][cj][ci];

824/**
825 * @brief Computes the trilinear Interpolatetion weights from the Interpolatetion coefficients.
826 *
827 * This function computes the weights for trilinear Interpolatetion at the eight corners of a cell
828 * using the Interpolatetion coefficients provided along the x, y, and z directions.
829 *
830 * @param[in] a1 Interpolation coefficient along the x-direction (normalized coordinate within the cell).
831 * @param[in] a2 Interpolation coefficient along the y-direction (normalized coordinate within the cell).
832 * @param[in] a3 Interpolation coefficient along the z-direction (normalized coordinate within the cell).
833 * @param[out] w Array of 8 weights, each corresponding to one corner of the cell.
834 *
835 * @note
836 * - The coefficients `a1`, `a2`, and `a3` should be in the range [0, 1].
837 * - The order of weights corresponds to the eight corners of a hexahedral cell.
838 */
839static inline void ComputeTrilinearWeights(PetscReal a1, PetscReal a2, PetscReal a3, PetscReal *w) {
840 LOG_ALLOW(GLOBAL, LOG_DEBUG, "Computing weights for a1=%f, a2=%f, a3=%f.\n", a1, a2, a3);
841
842 // Ensure a1, a2, a3 are within [0,1]
843 a1 = PetscMax(0.0, PetscMin(1.0, a1));
844 a2 = PetscMax(0.0, PetscMin(1.0, a2));
845 a3 = PetscMax(0.0, PetscMin(1.0, a3));
846
847 const PetscReal oa1 = 1.0 - a1;
848 const PetscReal oa2 = 1.0 - a2;
849 const PetscReal oa3 = 1.0 - a3;
850
851 w[0] = oa1 * oa2 * oa3; /* cornerOffsets[0] => (0,0,0) */
852 w[1] = a1 * oa2 * oa3; /* cornerOffsets[1] => (1,0,0) */
853 w[2] = oa1 * a2 * oa3; /* cornerOffsets[2] => (0,1,0) */
854 w[3] = a1 * a2 * oa3; /* cornerOffsets[3] => (1,1,0) */
855 w[4] = oa1 * oa2 * a3; /* cornerOffsets[4] => (0,0,1) */
856 w[5] = a1 * oa2 * a3; /* cornerOffsets[5] => (1,0,1) */
857 w[6] = oa1 * a2 * a3; /* cornerOffsets[6] => (0,1,1) */
858 w[7] = a1 * a2 * a3; /* cornerOffsets[7] => (1,1,1) */
859
860 // Log the computed weights for debugging
861 LOG_ALLOW(LOCAL,LOG_DEBUG, "Weights computed - "
862 "w0=%f, w1=%f, w2=%f, w3=%f, w4=%f, w5=%f, w6=%f, w7=%f. \n",
863 w[0], w[1], w[2], w[3], w[4], w[5], w[6], w[7]);
864}

868 *
869 * @param[in] fieldName A string representing the name of the scalar field (e.g., "temperature").

950/**
951 * @brief Computes the trilinear Interpolateted vector at a given point, with partial weighting if corners go out of range.
952 *
953 * If any of the 8 corners (i or i+1, j or j+1, k or k+1) is out of [0..mx), [0..my), [0..mz),
954 * that corner is skipped and the corresponding weight is omitted. The total is normalized
955 * by the sum of used weights, so we get a partial Interpolatetion near boundaries.
956 */
957PetscErrorCode TrilinearInterpolation_Vector(
958 const char *fieldName,
959 Cmpnts ***fieldVec, /* 3D array [k][j][i], dimension [mz][my][mx] */
960 PetscInt i, // local cell index i
961 PetscInt j, // local cell index j
962 PetscInt k, // local cell index k
963 PetscReal a1,
964 PetscReal a2,
965 PetscReal a3,

967{
968 PetscFunctionBegin; // PETSc macro for error/stack tracing

970 // Compute the 8 corner weights
971 PetscErrorCode ierr;

973 PetscMPIInt rank;

983 // and how much sum of weights is used. Then we do a final normalization.
984 PetscReal sumW = 0.0;
985 Cmpnts accum = {0.0, 0.0, 0.0};
986
987 // The eight corner indices, with their weights:
988 // corners: (i,j,k), (i+1,j,k), (i,j+1,k), (i+1,j+1,k), etc.
989 // We store them in an array to iterate cleanly.
990 const PetscInt cornerOffsets[8][3] = {
991 {0, 0, 0},
992 {1, 0, 0},
993 {0, 1, 0},
994 {1, 1, 0},
995 {0, 0, 1},
996 {1, 0, 1},
997 {0, 1, 1},
998 {1, 1, 1}
999 };

1002 for (PetscInt c = 0; c < 8; c++) {
1003 const PetscInt di = cornerOffsets[c][0];
1004 const PetscInt dj = cornerOffsets[c][1];
1005 const PetscInt dk = cornerOffsets[c][2];

1007 PetscInt jC = j + dj;
1008 PetscInt kC = k + dk;
1009

1012 // (Assuming you know global domain is [0..mx), [0..my), [0..mz).)

1014 jC < 0 || jC >= (PetscInt)userGlobalMy ||

1017 // skip this corner
1018 continue;
1019 }

1021 */

1024
1025 // Otherwise, accumulate
1026 accum.x += wcorner[c] * fieldVec[kC][jC][iC].x;
1027 accum.y += wcorner[c] * fieldVec[kC][jC][iC].y;
1028 accum.z += wcorner[c] * fieldVec[kC][jC][iC].z;
1029 sumW += wcorner[c];
1030 }
1031
1032 // If sumW=0 => out-of-range or zero weighting => set (0,0,0)
1033 if (sumW > 1.0e-14) {
1034 vec->x = accum.x / sumW;
1035 vec->y = accum.y / sumW;
1036 vec->z = accum.z / sumW;
1037 } else {
1038 vec->x = 0.0; vec->y = 0.0; vec->z = 0.0;
1039 }
1040
1041 PetscFunctionReturn(0);
