24/**
25 * @brief Creates and initializes a Particle Swarm.
26 *
27 * This function sets up a DMSwarm within the provided UserCtx structure, initializes
28 * particle fields, and distributes particles across MPI processes. It ensures that
29 * the number of particles is evenly divided among the available MPI ranks. If the total
30 * number of particles isn't divisible by the number of processes, the remainder is distributed
31 * to the first few ranks.
32 *
33 * Additionally, it now takes a 'bboxlist' array as an input parameter and passes it on to
34 * AssignInitialProperties(), enabling particle initialization at the midpoint of each rank's
35 * bounding box if ParticleInitialization is set to 0.
36 *
37 * @param[in,out] user Pointer to the UserCtx structure containing the simulation context.
38 * @param[in] numParticles Total number of particles to create across all MPI processes.
39 * @param[in] bboxlist Pointer to an array of BoundingBox structures, one per rank.
40 *
41 * @return PetscErrorCode Returns 0 on success, non-zero on failure.
42 *
43 * @note
44 * - Ensure that `numParticles` is a positive integer.
45 * - The `control.dat` file should contain necessary PETSc options.
46 * - The `bboxlist` array should be properly populated before calling this function.
47 */
48PetscErrorCode CreateParticleSwarm(UserCtx *user, PetscInt numParticles, PetscInt *particlesPerProcess, BoundingBox *bboxlist);

102/**
103 * @brief Initializes all particle properties in the swarm.
104 *
105 * This function orchestrates the initialization of particle properties.
106 * It first determines the inlet face if surface initialization (Mode 0) is selected
107 * by parsing "bcs.dat".
108 * Then, it initializes basic particle properties (physical position, Particle ID,
109 * and placeholder Cell IDs) by calling `InitializeParticleBasicProperties`. This call
110 * uses the provided `rand_logic_i/j/k` RNGs, which must be pre-initialized for [0,1).
111 * The `rand_phys_x/y/z` RNGs (physically bounded) are passed but may not be used by
112 * `InitializeParticleBasicProperties` for position setting if all initialization paths
113 * use logical-to-physical mapping.
114 * Finally, it calls helper functions to initialize other registered swarm fields
115 * like "velocity", "weight", and "P" (pressure) to default values.
116 *
117 * @param[in,out] user Pointer to the `UserCtx` structure.
118 * @param[in] particlesPerProcess Number of particles assigned to this MPI process.
119 * @param[in] rand_phys_x RNG for physical x-coordinates (from `InitializeRandomGenerators`).
120 * @param[in] rand_phys_y RNG for physical y-coordinates (from `InitializeRandomGenerators`).
121 * @param[in] rand_phys_z RNG for physical z-coordinates (from `InitializeRandomGenerators`).
122 * @param[in] rand_logic_i RNG for i-logical dimension tasks [0,1) (from `InitializeLogicalSpaceRNGs`).
123 * @param[in] rand_logic_j RNG for j-logical dimension tasks [0,1) (from `InitializeLogicalSpaceRNGs`).
124 * @param[in] rand_logic_k RNG for k-logical dimension tasks [0,1) (from `InitializeLogicalSpaceRNGs`).
125 * @param[in] bboxlist Array of BoundingBox structures (potentially unused by IPBP).
126 *
127 * @return PetscErrorCode Returns 0 on success, non-zero on failure.
128 */

130 PetscInt particlesPerProcess,
131 PetscRandom *rand_phys_x, // RNG from original InitializeRandomGenerators
132 PetscRandom *rand_phys_y, // RNG from original InitializeRandomGenerators
133 PetscRandom *rand_phys_z, // RNG from original InitializeRandomGenerators
134 PetscRandom *rand_logic_i, // RNG from InitializeLogicalSpaceRNGs
135 PetscRandom *rand_logic_j, // RNG from InitializeLogicalSpaceRNGs

241/**
242 * @brief Resets the location-dependent state of a loaded swarm to force relocation.
243 * @ingroup ParticleRestart
244 *
245 * This function is a critical part of the simulation restart procedure. It must be
246 * called immediately after `ReadAllSwarmFields` has populated a swarm from restart
247 * files. Its purpose is to invalidate the "location" state of the loaded particles,
248 * ensuring that the `LocateAllParticlesInGrid_TEST` orchestrator performs a fresh,
249 * comprehensive search for every particle based on its loaded position.
250 *
251 * It does this by performing two actions on every locally-owned particle:
252 * 1. It resets the `DMSwarm_CellID` field to a sentinel value of `(-1, -1, -1)`.
253 * This invalidates any cell index that might have been loaded or defaulted to 0.
254 * 2. It sets the `DMSwarm_location_status` field to `NEEDS_LOCATION`.
255 *
256 * This guarantees that the location logic will not mistakenly use a stale cell index
257 * from a previous run and will instead use the robust "Guess -> Verify" strategy
258 * appropriate for particles with unknown locations.
259 *
260 * @param[in,out] user Pointer to the UserCtx structure which contains the `DMSwarm` object
261 * that has just been loaded with data from restart files.
262 * @return PetscErrorCode 0 on success, or a non-zero PETSc error code if field access fails.
263 */
264PetscErrorCode PrepareLoadedSwarmForRelocation(UserCtx *user);

266/**
267 * @brief Perform particle swarm initialization, particle-grid interaction, and related operations.
268 *
269 * This function handles the following tasks:
270 * 1. Initializes the particle swarm using the provided bounding box list (bboxlist) to determine initial placement
271 * if ParticleInitialization is 0.
272 * 2. Locates particles within the computational grid.
273 * 3. Updates particle positions based on grid interactions (if such logic exists elsewhere in the code).
274 * 4. Interpolates particle velocities from grid points using trilinear interpolation.
275 *
276 * @param[in,out] user Pointer to the UserCtx structure containing grid and particle swarm information.
277 * @param[in] np Number of particles to initialize in the swarm.
278 * @param[in] bboxlist Pointer to an array of BoundingBox structures, one per MPI rank.
279 *
280 * @return PetscErrorCode Returns 0 on success, non-zero on failure.
281 *
282 * @note
283 * - Ensure that `np` (number of particles) is positive.
284 * - The `bboxlist` array must be correctly computed and passed in before calling this function.
285 * - If ParticleInitialization == 0, particles will be placed at the midpoint of the local bounding box.
286 */
287PetscErrorCode InitializeParticleSwarm(SimCtx *simCtx);
