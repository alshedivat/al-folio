9/**
10 * @brief Allocates and populates the master SimulationContext object.
11 *
12 * This function serves as the single, authoritative entry point for all
13 * simulation configuration. It merges the setup logic from both the legacy
14 * FSI/IBM solver and the modern particle solver into a unified, robust
15 * process.
16 *
17 * The function follows a strict sequence:
18 * 1. **Allocate Context & Set Defaults:** It first allocates the `SimulationContext`
19 * and populates every field with a sane, hardcoded default value. This
20 * ensures the simulation starts from a known, predictable state.
21 * 2. **Configure Logging System:** It configures the custom logging framework. It
22 * parses the `-func_config_file` option to load a list of function names
23 * allowed to produce log output. This configuration (the file path and the
24 * list of function names) is stored within the `SimulationContext` for
25 * later reference and cleanup.
26 * 3. **Parse All Options:** It performs a comprehensive pass of `PetscOptionsGet...`
27 * calls for every possible command-line flag, overriding the default
28 * values set in step 1.
29 * 4. **Log Summary:** After all options are parsed, it uses the now-active
30 * logging system to print a summary of the key simulation parameters.
31 *
32 * @param[in] argc Argument count passed from `main`.
33 * @param[in] argv Argument vector passed from `main`.
34 * @param[out] p_simCtx On success, this will point to the newly created and
35 * fully configured `SimulationContext` pointer. The caller
36 * is responsible for eventually destroying this object by
37 * calling `FinalizeSimulation()`.
38 *
39 * @return PetscErrorCode Returns 0 on success, or a non-zero PETSc error code on failure.
40 */

405/**
406 * @brief Allocates the memory for the UserMG and UserCtx hierarchy.
407 *
408 * This function performs the foundational memory allocation for the solver's
409 * data structures. It reads the number of multigrid levels and grid blocks
410 * from the master SimulationContext, then allocates the arrays for MGCtx and
411 * UserCtx.
412 *
413 * It performs three critical tasks:
414 * 1. Allocates the MGCtx array within the UserMG struct.
415 * 2. For each level, allocates the UserCtx array for all blocks.
416 * 3. Sets the `simCtx` back-pointer in every single UserCtx, linking it to the
417 * master configuration.
418 *
419 * @param simCtx The master SimulationContext, which contains configuration and
420 * will store the resulting allocated hierarchy in its `usermg` member.
421 * @return PetscErrorCode
422 */
423static PetscErrorCode AllocateContextHierarchy(SimCtx *simCtx)
424{
425 PetscErrorCode ierr;
426 UserMG *usermg = &simCtx->usermg;
427 MGCtx *mgctx;
428 PetscInt nblk = simCtx->block_number;
429 PetscBool found;
430 PetscFunctionBeginUser;
431
432 LOG_ALLOW(GLOBAL, LOG_INFO, "Allocating context hierarchy for %d levels and %d blocks...\n", simCtx->mglevels, nblk);
433
434 // Store the number of levels in the UserMG struct itself
435 usermg->mglevels = simCtx->mglevels;
436
437 // --- 1. Allocate the array of MGCtx structs ---
438 ierr = PetscMalloc(usermg->mglevels * sizeof(MGCtx), &usermg->mgctx); CHKERRQ(ierr);
439 mgctx = usermg->mgctx;
440 LOG_ALLOW(LOCAL, LOG_DEBUG, "Rank %d: Allocated MGCtx array of size %d.\n", simCtx->rank, usermg->mglevels);
441
442 // --- 2. Parse semi-coarsening options (logic from MG_Initial) ---
443 // These flags determine if a dimension is coarsened in the multigrid hierarchy.
444 PetscInt *isc, *jsc, *ksc;
445 ierr = PetscMalloc3(nblk, &isc, nblk, &jsc, nblk, &ksc); CHKERRQ(ierr);
446 // Set defaults to FALSE (full coarsening)
447 for (PetscInt i = 0; i < nblk; ++i) {
448 isc[i] = 0; jsc[i] = 0; ksc[i] = 0;
449 }
450
451// Use a temporary variable for the 'count' argument to the parsing function.
452 // This protects the original 'nblk' which is needed for the loop bounds.
453 PetscInt n_opts_found = nblk;
454 ierr = PetscOptionsGetIntArray(NULL, NULL, "-mg_i_semi", isc, &n_opts_found, &found); CHKERRQ(ierr);
455
456 n_opts_found = nblk; // Reset the temp variable before the next call
457 ierr = PetscOptionsGetIntArray(NULL, NULL, "-mg_j_semi", jsc, &n_opts_found, &found); CHKERRQ(ierr);
458
459 n_opts_found = nblk; // Reset the temp variable before the next call
460 ierr = PetscOptionsGetIntArray(NULL, NULL, "-mg_k_semi", ksc, &n_opts_found, &found); CHKERRQ(ierr);
461
462 // --- 3. Loop over levels and blocks to allocate UserCtx arrays ---
463 for (PetscInt level = 0; level < simCtx->mglevels; level++) {
464
465 LOG_ALLOW(LOCAL, LOG_DEBUG, "Rank %d: Setting up MG Level %d...\n", simCtx->rank, level);
466 // Allocate the array of UserCtx structs for this level
467 ierr = PetscMalloc(nblk * sizeof(UserCtx), &mgctx[level].user); CHKERRQ(ierr);
468 // It's good practice to zero out the memory to avoid uninitialized values
469 ierr = PetscMemzero(mgctx[level].user, nblk * sizeof(UserCtx)); CHKERRQ(ierr);
470 mgctx[level].thislevel = level;
471
472 for (PetscInt bi = 0; bi < nblk; bi++) {
473 UserCtx *currentUser = &mgctx[level].user[bi];
474 LOG_ALLOW(LOCAL, LOG_DEBUG, "Rank %d: Initializing UserCtx for Level %d, Block %d.\n", simCtx->rank, level, bi);
475
476 // --- CRITICAL STEP: Set the back-pointer to the master context ---
477 currentUser->simCtx = simCtx;
478
479 // Initialize other per-context values
480 currentUser->thislevel = level;
481 currentUser->_this = bi; // Store the block index
482 currentUser->mglevels = usermg->mglevels;
483
484 // Assign semi-coarsening flags
485 currentUser->isc = isc[bi];
486 currentUser->jsc = jsc[bi];
487 currentUser->ksc = ksc[bi];
488
489 // Link to finer/coarser contexts for multigrid operations
490 if (level > 0) {
491 currentUser->user_c = &mgctx[level-1].user[bi];
492 LOG_ALLOW_SYNC(GLOBAL, LOG_DEBUG, "Rank %d: -> Linked to coarser context (user_c).\n", simCtx->rank);
493 }
494 if (level < usermg->mglevels - 1) {
495 currentUser->user_f = &mgctx[level+1].user[bi];
496 LOG_ALLOW_SYNC(GLOBAL, LOG_DEBUG, "Rank %d: -> Linked to finer context (user_f).\n", simCtx->rank);
497 }
498 }
499 }
500
501 // Log a summary of the parsed flags on each rank.
502 if (get_log_level() >= LOG_DEBUG && nblk > 0) {
503 LOG_ALLOW(LOCAL, LOG_DEBUG, "Rank %d: Final semi-coarsening configuration view:\n", simCtx->rank);
504 for (PetscInt bi = 0; bi < nblk; ++bi) {
505 LOG_ALLOW(LOCAL, LOG_DEBUG, "Rank %d: Block %d: i-semi=%d, j-semi=%d, k-semi=%d\n", simCtx->rank, bi, isc[bi], jsc[bi], ksc[bi]);
506 }
507 }
508
509 // Clean up temporary arrays
510 ierr = PetscFree3(isc, jsc, ksc); CHKERRQ(ierr);
511
512 LOG_ALLOW(GLOBAL, LOG_INFO, "Context hierarchy allocation complete.\n");
513 PetscFunctionReturn(0);
514}

1007/**
1008 * @brief Allocates a 3D array of PetscReal values using PetscCalloc.
1009 *
1010 * This function dynamically allocates memory for a 3D array of PetscReal values
1011 * with dimensions nz (layers) x ny (rows) x nx (columns). It uses PetscCalloc1
1012 * to ensure the memory is zero-initialized.
1013 *
1014 * The allocation is done in three steps:
1015 * 1. Allocate an array of nz pointers (one for each layer).
1016 * 2. Allocate a contiguous block for nz*ny row pointers and assign each layers row pointers.
1017 * 3. Allocate a contiguous block for all nz*ny*nx PetscReal values.
1018 *
1019 * This setup allows the array to be accessed as array[k][j][i], and the memory
1020 * for the data is contiguous, which improves cache efficiency.
1021 *
1022 * @param[out] array Pointer to the 3D array to be allocated.
1023 * @param[in] nz Number of layers (z-direction).
1024 * @param[in] ny Number of rows (y-direction).
1025 * @param[in] nx Number of columns (x-direction).
1026 *
1027 * @return PetscErrorCode 0 on success, nonzero on failure.
1028 */
1029PetscErrorCode Allocate3DArrayScalar(PetscReal ****array, PetscInt nz, PetscInt ny, PetscInt nx)
1030{
1031 PetscErrorCode ierr;
1032 PetscReal ***data;
1033 PetscReal *dataContiguous;
1034 PetscInt k, j;
1035
1036 PetscFunctionBegin;
1037 /* Step 1: Allocate memory for an array of nz layer pointers (zero-initialized) */
1038 ierr = PetscCalloc1(nz, &data); CHKERRQ(ierr);
1039
1040 /* Step 2: Allocate memory for all row pointers (nz * ny pointers) */
1041 ierr = PetscCalloc1(nz * ny, &data[0]); CHKERRQ(ierr);
1042 for (k = 1; k < nz; k++) {
1043 data[k] = data[0] + k * ny;
1044 }
1045
1046 /* Step 3: Allocate one contiguous block for all data elements (nz*ny*nx) */
1047 ierr = PetscCalloc1(nz * ny * nx, &dataContiguous); CHKERRQ(ierr);
1048
1049 /* Build the 3D pointer structure: each row pointer gets the correct segment of data */
1050 for (k = 0; k < nz; k++) {
1051 for (j = 0; j < ny; j++) {
1052 data[k][j] = dataContiguous + (k * ny + j) * nx;
1053 /* Memory is already zeroed by PetscCalloc1, so no manual initialization is needed */
1054 }
1055 }
1056 *array = data;
1057 PetscFunctionReturn(0);
1058}

1187 * @return PetscErrorCode 0 on success, nonzero on failure.
1188 */
1189 PetscErrorCode Deallocate3DArrayVector(Cmpnts ***array, PetscInt nz, PetscInt ny)
1190{
1191 PetscErrorCode ierr;
1192
1193 PetscFunctionBegin;
1194 // If array is NULL or hasn't been allocated properly, just return.
1195 if (!array || !array[0] || !array[0][0] ) {
1196 LOG_ALLOW(GLOBAL, LOG_WARNING, "Deallocate3DArrayVector called with potentially unallocated or NULL array.\n");
1197 // Attempt to free what might exist, but be cautious
1198 if (array) {
1199 if (array[0]) { // Check if row pointers were allocated
1200 // We don't have a direct pointer to the contiguous data block
1201 // saved separately in this allocation scheme. The allocation relies
1202 // on array[0][0] pointing to it. If array[0] was freed first,
1203 // accessing array[0][0] is unsafe.
1204 // The allocation scheme where the contiguous data block is not
1205 // stored separately makes safe deallocation tricky if freeing
1206 // happens out of order or if parts are NULL.
1207
1208 // A SAFER ALLOCATION/DEALLOCATION would store the data pointer separately.
1209 // Given the current allocation scheme, the order MUST be:
1210 // 1. Free the data block (pointed to by array[0][0])
1211 // 2. Free the row pointer block (pointed to by array[0])
1212 // 3. Free the layer pointer block (pointed to by array)
1213
1214 // Let's assume the allocation was successful and pointers are valid.
1215 // Get pointer to the contiguous data block *before* freeing row pointers
1216 Cmpnts *dataContiguous = array[0][0];
1217 ierr = PetscFree(dataContiguous); CHKERRQ(ierr); // Free data block
1218
1219 // Now free the row pointers block
1220 ierr = PetscFree(array[0]); CHKERRQ(ierr); // Free row pointers
1221
1222 }
1223 // Finally, free the array of layer pointers
1224 ierr = PetscFree(array); CHKERRQ(ierr);
1225 }
1226 PetscFunctionReturn(0); // Return gracefully if input was NULL initially
1227 }
1228
1229
1230 // --- Standard Deallocation (assuming valid allocation) ---
1231
1232 /* 1. Free the contiguous block of Cmpnts structures.
1233 The starting address was stored in array[0][0] by Allocate3DArrayVector. */
1234 ierr = PetscFree(array[0][0]); CHKERRQ(ierr); // Free the ACTUAL DATA
1235
1236 /* 2. Free the contiguous block of row pointers.
1237 The starting address was stored in array[0]. */
1238 ierr = PetscFree(array[0]); CHKERRQ(ierr); // Free the ROW POINTERS
1239
1240 /* 3. Free the layer pointer array.
1241 The starting address is 'array' itself. */
1242 ierr = PetscFree(array); CHKERRQ(ierr); // Free the LAYER POINTERS
1243
1244 PetscFunctionReturn(0);
1245}

1247/**
1248 * @brief Gets the global starting index of cells owned by this rank and the number of such cells.
1249 *
1250 * A cell's global index is considered the same as its origin node's global index.
1251 * This function assumes a node-centered DMDA where `info_nodes` provides all necessary
1252 * information:
1253 * - `info_nodes->xs, ys, zs`: Global starting index of the first node owned by this rank (excluding ghosts).
1254 * - `info_nodes->xm, ym, zm`: Number of nodes owned by this rank in each dimension (excluding ghosts).
1255 * - `info_nodes->mx, my, mz`: Total number of global nodes in each dimension for the entire domain.
1256 *
1257 * A cell `C_k` (0-indexed) is defined by its origin node `N_k` and extends to node `N_{k+1}`.
1258 * Thus, the last node in the global domain cannot be an origin for a cell. The last possible
1259 * cell origin node index is `GlobalNodesInDim - 2`.
1260 *
1261 * @param[in] info_nodes Pointer to the DMDALocalInfo struct for the current rank.
1262 * This struct contains local ownership information (xs, xm, etc.)
1263 * and global domain dimensions (mx, my, mz for nodes).
1264 * @param[in] dim The dimension for which to get the cell range (0 for i/x, 1 for j/y, 2 for k/z).
1265 * @param[out] xs_cell_global_out Pointer to store the global index of the first cell whose origin node
1266 * is owned by this rank. If the rank owns no valid cell origins in this
1267 * dimension, this will be the rank's starting node index, but
1268 * `xm_cell_local_out` will be 0.
1269 * @param[out] xm_cell_local_out Pointer to store the number of cells for which this rank owns the
1270 * origin node AND that origin node is a valid cell origin within the
1271 * global domain.
1272 *
1273 * @return PetscErrorCode 0 on success, or an error code on failure.
1274 *
1275 * @note Example: If GlobalNodesInDim = 11 (nodes N0 to N10), there are 10 cells (C0 to C9).
1276 * The last cell, C9, has its origin at node N9. So, N9 (index 9) is the last valid
1277 * cell origin (GlobalNodesInDim - 2 = 11 - 2 = 9).
1278 * If a rank owns nodes N8, N9, N10 (xs=8, xm=3):
1279 * - First potential origin on rank = N8.
1280 * - Last potential origin on rank (node that is not the last owned node) = N9.
1281 * - Actual last origin this rank can form = min(N9, GlobalMaxOrigin=N9) = N9.
1282 * - Number of cells = (N9 - N8 + 1) = 2 cells (C8, C9).
1283 * If a rank owns only node N10 (xs=10, xm=1):
1284 * - First potential origin on rank = N10.
1285 * - Actual last origin rank can form = min(N9, GlobalMaxOrigin=N9) (since N10-1=N9).
1286 * - first_potential_origin_on_rank (N10) > actual_last_origin_this_rank_can_form (N9) => 0 cells.
1287 */
1288PetscErrorCode GetOwnedCellRange(const DMDALocalInfo *info_nodes,
1289 PetscInt dim,
1290 PetscInt *xs_cell_global_out,
1291 PetscInt *xm_cell_local_out)
1292{
1293 PetscErrorCode ierr = 0; // Standard PETSc error code, not explicitly set here but good practice.
1294 PetscInt xs_node_global_rank; // Global index of the first node owned by this rank in the specified dimension.
1295 PetscInt num_nodes_owned_rank; // Number of nodes owned by this rank in this dimension (local count, excluding ghosts).
1296 PetscInt GlobalNodesInDim_from_info; // Total number of global nodes in this dimension, from DMDALocalInfo.
1297
1298 PetscFunctionBeginUser;
1299
1300 // --- 1. Input Validation ---
1301 if (!info_nodes || !xs_cell_global_out || !xm_cell_local_out) {
1302 SETERRQ(PETSC_COMM_SELF, PETSC_ERR_ARG_NULL, "Null pointer passed to GetOwnedCellRange.");
1303 }
1304
1305 // --- 2. Extract Node Ownership and Global Dimension Information from DMDALocalInfo ---
1306 if (dim == 0) { // I-direction
1307 xs_node_global_rank = info_nodes->xs; // Starting owned node index (global)
1308 num_nodes_owned_rank = info_nodes->xm; // Number of nodes owned by this rank (local count)
1309 GlobalNodesInDim_from_info = info_nodes->mx; // Total global nodes in this dimension
1310 } else if (dim == 1) { // J-direction
1311 xs_node_global_rank = info_nodes->ys;
1312 num_nodes_owned_rank = info_nodes->ym;
1313 GlobalNodesInDim_from_info = info_nodes->my;
1314 } else if (dim == 2) { // K-direction
1315 xs_node_global_rank = info_nodes->zs;
1316 num_nodes_owned_rank = info_nodes->zm;
1317 GlobalNodesInDim_from_info = info_nodes->mz;
1318 } else {
1319 SETERRQ(PETSC_COMM_SELF, PETSC_ERR_ARG_OUTOFRANGE, "Invalid dimension %d in GetOwnedCellRange. Must be 0, 1, or 2.", dim);
1320 }
1321
1322 // --- 3. Handle Edge Cases for Global Domain Size ---
1323 // If the global domain has 0 or 1 node plane, no cells can be formed.
1324 if (GlobalNodesInDim_from_info <= 1) {
1325 *xs_cell_global_out = xs_node_global_rank; // Still report the rank's starting node
1326 *xm_cell_local_out = 0; // But 0 cells
1327 PetscFunctionReturn(0);
1328 }
1329 // Negative global dimension is an error (should be caught by DMDA setup, but defensive)
1330 if (GlobalNodesInDim_from_info < 0 ) {
1331 SETERRQ(PETSC_COMM_SELF, PETSC_ERR_ARG_OUTOFRANGE, "GlobalNodesInDim %d from DMDALocalInfo must be non-negative for dimension %d.", GlobalNodesInDim_from_info, dim);
1332 }
1333
1334 // --- 4. Determine Cell Ownership Based on Node Ownership ---
1335 // The first cell this rank *could* define has its origin at the first node this rank owns.
1336 *xs_cell_global_out = xs_node_global_rank;
1337
1338 // If the rank owns no nodes in this dimension, it can't form any cell origins.
1339 if (num_nodes_owned_rank == 0) {
1340 *xm_cell_local_out = 0;
1341 } else {
1342 // Calculate the global index of the last possible node that can serve as a cell origin.
1343 // If GlobalNodesInDim = N (nodes 0 to N-1), cells are C_0 to C_{N-2}.
1344 // The origin of cell C_{N-2} is node N_{N-2}.
1345 // So, the last valid cell origin node index is (GlobalNodesInDim - 2).
1346 PetscInt last_possible_origin_global_idx = GlobalNodesInDim_from_info - 2;
1347
1348 // Determine the range of nodes owned by this rank that could *potentially* be cell origins.
1349 // The first node owned by the rank is a potential origin.
1350 PetscInt first_potential_origin_on_rank = xs_node_global_rank;
1351
1352 // A node can be an origin if there's at least one node after it to form the cell.
1353 // So, the last node owned by the rank that could *potentially* be an origin is
1354 // the second-to-last node it owns: (xs_node_global_rank + num_nodes_owned_rank - 1) - 1
1355 // which simplifies to: xs_node_global_rank + num_nodes_owned_rank - 2.
1356 PetscInt last_potential_origin_on_rank = xs_node_global_rank + num_nodes_owned_rank - 2;
1357
1358 // The actual last origin this rank can provide is capped by the global domain limit.
1359 PetscInt actual_last_origin_this_rank_can_form = PetscMin(last_potential_origin_on_rank, last_possible_origin_global_idx);
1360
1361 // If the first potential origin this rank owns is already beyond the actual last origin it can form,
1362 // then this rank forms no valid cell origins. This happens if:
1363 // - num_nodes_owned_rank is 1 (so last_potential_origin_on_rank = first_potential_origin_on_rank - 1).
1364 // - The rank only owns nodes at the very end of the global domain (e.g., only the last global node).
1365 if (first_potential_origin_on_rank > actual_last_origin_this_rank_can_form) {
1366 *xm_cell_local_out = 0;
1367 } else {
1368 // The number of cells is the count of valid origins this rank owns.
1369 // (Count = Last Index - First Index + 1)
1370 *xm_cell_local_out = actual_last_origin_this_rank_can_form - first_potential_origin_on_rank + 1;
1371 }
1372 }
1373 PetscFunctionReturn(ierr);
1374}

1376/**
1377 * @brief Gets the global cell range for a rank, including boundary cells.
1378 *
1379 * This function first calls GetOwnedCellRange to get the conservative range of
1380 * fully-contained cells. It then extends this range by applying the
1381 * "Lower-Rank-Owns-Boundary" principle. A rank claims ownership of the
1382 * boundary cells it shares with neighbors in the positive (+x, +y, +z)
1383 * directions.
1384 *
1385 * This results in a final cell range that is gap-free and suitable for building
1386 * the definitive particle ownership map.
1387 *
1388 * @param[in] info_nodes Pointer to the DMDALocalInfo struct.
1389 * @param[in] neighbors Pointer to the RankNeighbors struct containing neighbor info.
1390 * @param[in] dim The dimension (0 for i/x, 1 for j/y, 2 for k/z).
1391 * @param[out] xs_cell_global_out Pointer to store the final starting cell index.
1392 * @param[out] xm_cell_local_out Pointer to store the final number of cells.
1393 *
1394 * @return PetscErrorCode 0 on success, or an error code on failure.
1395 */
1396PetscErrorCode GetGhostedCellRange(const DMDALocalInfo *info_nodes,
1397 const RankNeighbors *neighbors,
1398 PetscInt dim,
1399 PetscInt *xs_cell_global_out,
1400 PetscInt *xm_cell_local_out)
1401{
1402 PetscErrorCode ierr;
1403
1404 PetscFunctionBeginUser;
1405
1406 // --- 1. Get the base, conservative range from the original function ---
1407 ierr = GetOwnedCellRange(info_nodes, dim, xs_cell_global_out, xm_cell_local_out); CHKERRQ(ierr);
1408
1409 // --- 2. Apply the "Lower-Rank-Owns-Boundary" correction ---
1410 // A rank owns the boundary if it has a neighbor in the positive direction.
1411 // We check if the neighbor's rank is valid (not MPI_PROC_NULL, which is < 0).
1412 if (dim == 0 && neighbors->rank_xp > -1) {
1413 (*xm_cell_local_out)++;
1414 } else if (dim == 1 && neighbors->rank_yp > -1) {
1415 (*xm_cell_local_out)++;
1416 } else if (dim == 2 && neighbors->rank_zp > -1) {
1417 (*xm_cell_local_out)++;
1418 }
1419
1420 PetscFunctionReturn(0);

1444

1500 user->neighbors.rank_xp = temp_neighbor;

1623 * existing helper functions to sequentially gather and broadcast the bounding box
1624 * information for each computational block. All ranks participate in each step,
1625 * assembling the final, unified multi-block list in parallel.
1626 *
1627 * @param[in,out] simCtx The master simulation context. After this call,
1628 * simCtx->bboxlist and the relevant user->neighbors fields
1629 * will be populated on all ranks.
1630 */
1631PetscErrorCode SetupDomainRankInfo(SimCtx *simCtx)
1632{
1633 PetscErrorCode ierr;
1634 PetscInt nblk = simCtx->block_number;
1635 PetscInt size = simCtx->size;
1636 BoundingBox *final_bboxlist = NULL;
1637
1638 PetscFunctionBeginUser;
1639
1640 LOG_ALLOW(GLOBAL, LOG_INFO, "Starting full rank communication setup for %d block(s).\n", nblk);
1641
1642 UserCtx *user_finest = simCtx->usermg.mgctx[simCtx->usermg.mglevels - 1].user;
1643
1644 // --- Step 1: Compute neighbor ranks (unchanged) ---
1645 for (int bi = 0; bi < nblk; bi++) {
1646 ierr = ComputeAndStoreNeighborRanks(&user_finest[bi]); CHKERRQ(ierr);
1647 }
1648 LOG_ALLOW(GLOBAL, LOG_INFO, "Neighbor ranks computed and stored for all blocks.\n");
1649
1650 // --- Step 2: Allocate the final, unified list on ALL ranks ---
1651 // Every rank will build this list in parallel.
1652 ierr = PetscMalloc1(size * nblk, &final_bboxlist); CHKERRQ(ierr);
1653
1654 // --- Step 3: Loop through each block, gather then broadcast its bbox list ---
1655 for (int bi = 0; bi < nblk; bi++) {
1656 // This is a temporary pointer for the current block's list.
1657 BoundingBox *block_bboxlist = NULL;
1658
1659 LOG_ALLOW(GLOBAL, LOG_INFO, "Processing bounding boxes for block %d...\n", bi);
1660
1661 // A) GATHER: On rank 0, block_bboxlist is allocated and filled. On others, it's NULL.
1662 ierr = GatherAllBoundingBoxes(&user_finest[bi], &block_bboxlist); CHKERRQ(ierr);
1663 LOG_ALLOW(GLOBAL, LOG_DEBUG, " -> Gather complete for block %d.", bi);
1664
1665 // B) BROADCAST: On non-root ranks, block_bboxlist is allocated. Then, the data
1666 // from rank 0 is broadcast to all ranks. After this call, ALL ranks have
1667 // an identical, complete copy of the bounding boxes for the current block.
1668 ierr = BroadcastAllBoundingBoxes(&user_finest[bi], &block_bboxlist); CHKERRQ(ierr);
1669 LOG_ALLOW(GLOBAL, LOG_DEBUG, " -> Broadcast complete for block %d.\n", bi);
1670
1671 // C) ASSEMBLE: Every rank now copies the data for this block into the
1672 // correct segment of its final, unified list.
1673 for (int r = 0; r < size; r++) {
1674 // The layout is [r0b0, r1b0, ..., r(size-1)b0, r0b1, r1b1, ...]
1675 final_bboxlist[bi * size + r] = block_bboxlist[r];
1676 }
1677 LOG_ALLOW(GLOBAL, LOG_DEBUG, " -> Assembly into final list complete for block %d.\n", bi);
1678
1679 // D) CLEANUP: Free the temporary list for this block on ALL ranks before the next iteration.
1680 // Your helper functions use malloc, so we must use free.
1681 free(block_bboxlist);
1682 }
1683
1684 // --- Step 4: Assign the final pointer and run the last setup step ---
1685 simCtx->bboxlist = final_bboxlist;
1686 LOG_ALLOW(GLOBAL, LOG_INFO, "Final unified bboxlist created on all ranks and stored in SimCtx.\n");
1687
1688 ierr = SetupDomainCellDecompositionMap(&user_finest[0]); CHKERRQ(ierr);
1689 LOG_ALLOW(GLOBAL, LOG_INFO, "Domain Cell Composition set and broadcasted.\n");
1690
1691 LOG_ALLOW(GLOBAL, LOG_INFO, "SetupDomainRankInfo: Completed successfully.\n");
1692 PetscFunctionReturn(0);
1693}

1695/**
1696 * @brief Reconstructs Cartesian velocity (Ucat) at cell centers from contravariant
1697 * velocity (Ucont) defined on cell faces.
1698 *
1699 * This function performs the transformation from a contravariant velocity representation
1700 * (which is natural on a curvilinear grid) to a Cartesian (x,y,z) representation.
1701 * For each interior computational cell owned by the rank, it performs the following:
1702 *
1703 * 1. It averages the contravariant velocity components (U, U, U) from the
1704 * surrounding faces to get an estimate of the contravariant velocity at the cell center.
1705 * 2. It averages the metric vectors (Csi, Eta, Zet) from the surrounding faces
1706 * to get an estimate of the metric tensor at the cell center. This tensor forms
1707 * the transformation matrix.
1708 * 3. It solves the linear system `[MetricTensor] * [ucat] = [ucont]` for the
1709 * Cartesian velocity vector `ucat = (u,v,w)` using Cramer's rule.
1710 * 4. The computed Cartesian velocity is stored in the global `user->Ucat` vector.
1711 *
1712 * The function operates on local, ghosted versions of the input vectors (`user->lUcont`,
1713 * `user->lCsi`, etc.) to ensure stencils are valid across processor boundaries.
1714 *
1715 * @param[in,out] user Pointer to the UserCtx structure. The function reads from
1716 * `user->lUcont`, `user->lCsi`, `user->lEta`, `user->lZet`, `user->lNvert`
1717 * and writes to the global `user->Ucat` vector.
1718 *
1719 * @return PetscErrorCode 0 on success.
1720 *
1721 * @note
1722 * - This function should be called AFTER `user->lUcont` and all local metric vectors
1723 * (`user->lCsi`, etc.) have been populated with up-to-date ghost values via `UpdateLocalGhosts`.
1724 * - It only computes `Ucat` for interior cells (not on physical boundaries) and for
1725 * cells not marked as solid/blanked by `user->lNvert`.
1726 * - The caller is responsible for subsequently applying boundary conditions to `user->Ucat`
1727 * and calling `UpdateLocalGhosts(user, "Ucat")` to populate `user->lUcat`.
1728 */
1729PetscErrorCode Contra2Cart(UserCtx *user)
1730{
1731 PetscErrorCode ierr;
1732 DMDALocalInfo info;
1733 Cmpnts ***lcsi_arr, ***leta_arr, ***lzet_arr; // Local metric arrays
1734 Cmpnts ***lucont_arr; // Local contravariant velocity array
1735 Cmpnts ***gucat_arr; // Global Cartesian velocity array
1736 PetscReal ***lnvert_arr; // Local Nvert array
1737 PetscReal ***laj_arr; // Local Jacobian Determinant inverse array
1738
1739 PetscFunctionBeginUser;
1740 LOG_ALLOW(GLOBAL, LOG_DEBUG, "Starting Contravariant-to-Cartesian velocity transformation.\n");
1741
1742 // --- 1. Get DMDA Info and Check for Valid Inputs ---
1743 // All inputs (lUcont, lCsi, etc.) and outputs (Ucat) are on DMs from the UserCtx.
1744 // We get local info from fda, which governs the layout of most arrays here.
1745 ierr = DMDAGetLocalInfo(user->fda, &info); CHKERRQ(ierr);
1746 if (!user->lUcont || !user->lCsi || !user->lEta || !user->lZet || !user->lNvert || !user->Ucat) {
1747 SETERRQ(PETSC_COMM_SELF, PETSC_ERR_ARG_WRONGSTATE, "Contra2Cart requires lUcont, lCsi/Eta/Zet, lNvert, and Ucat to be non-NULL.");
1748 }
1749
1750
1751 // --- 2. Get Read-Only Array Access to Local Input Vectors (with ghosts) ---
1752 ierr = DMDAVecGetArrayRead(user->fda, user->lUcont, &lucont_arr); CHKERRQ(ierr);
1753 ierr = DMDAVecGetArrayRead(user->fda, user->lCsi, &lcsi_arr); CHKERRQ(ierr);
1754 ierr = DMDAVecGetArrayRead(user->fda, user->lEta, &leta_arr); CHKERRQ(ierr);
1755 ierr = DMDAVecGetArrayRead(user->fda, user->lZet, &lzet_arr); CHKERRQ(ierr);
1756 ierr = DMDAVecGetArrayRead(user->da, user->lNvert, &lnvert_arr); CHKERRQ(ierr);
1757 ierr = DMDAVecGetArrayRead(user->da, user->lAj, &laj_arr); CHKERRQ(ierr);
1758
1759 // --- 3. Get Write-Only Array Access to the Global Output Vector ---
1760 // We compute for local owned cells and write into the global vector.
1761 // PETSc handles mapping the global indices to the correct local memory locations.
1762 ierr = DMDAVecGetArray(user->fda, user->Ucat, &gucat_arr); CHKERRQ(ierr);
1763
1764
1765 // --- 4. Define Loop Bounds for INTERIOR Cells ---
1766 // We use adjusted bounds to avoid calculating Ucat on the physical domain boundaries,
1767 // as these are typically set explicitly by boundary condition functions.
1768 // The stencils use indices like i-1, j-1, k-1, so we must start loops at least at index 1.
1769 PetscInt i_start = (info.xs == 0) ? info.xs + 1 : info.xs;
1770 PetscInt i_end = (info.xs + info.xm == info.mx) ? info.xs + info.xm - 1 : info.xs + info.xm;
1771
1772 PetscInt j_start = (info.ys == 0) ? info.ys + 1 : info.ys;
1773 PetscInt j_end = (info.ys + info.ym == info.my) ? info.ys + info.ym - 1 : info.ys + info.ym;
1774
1775 PetscInt k_start = (info.zs == 0) ? info.zs + 1 : info.zs;
1776 PetscInt k_end = (info.zs + info.zm == info.mz) ? info.zs + info.zm - 1 : info.zs + info.zm;
1777
1778 // --- 5. Main Computation Loop ---
1779 // Loops over the GLOBAL indices of interior cells owned by this rank.
1780 for (PetscInt k_cell = k_start; k_cell < k_end; ++k_cell) {
1781 for (PetscInt j_cell = j_start; j_cell < j_end; ++j_cell) {
1782 for (PetscInt i_cell = i_start; i_cell < i_end; ++i_cell) {
1783
1784 // Check if the cell is a fluid cell (not solid/blanked)
1785 // if (lnvert_arr[k_cell][j_cell][i_cell] > 0.1) continue; // Skip solid/blanked cells
1786
1787 // Transformation matrix [mat] is the metric tensor at the cell center,
1788 // estimated by averaging metrics from adjacent faces.
1789 PetscReal mat[3][3];
1790
1791 // PetscReal aj_center = laj_arr[k_cell+1][j_cell+1][i_cell+1];
1792
1793 mat[0][0] = 0.5 * (lcsi_arr[k_cell][j_cell][i_cell-1].x + lcsi_arr[k_cell][j_cell][i_cell].x); //* aj_center;
1794 mat[0][1] = 0.5 * (lcsi_arr[k_cell][j_cell][i_cell-1].y + lcsi_arr[k_cell][j_cell][i_cell].y); //* aj_center;
1795 mat[0][2] = 0.5 * (lcsi_arr[k_cell][j_cell][i_cell-1].z + lcsi_arr[k_cell][j_cell][i_cell].z); //* aj_center;
1796
1797 mat[1][0] = 0.5 * (leta_arr[k_cell][j_cell-1][i_cell].x + leta_arr[k_cell][j_cell][i_cell].x); //* aj_center;
1798 mat[1][1] = 0.5 * (leta_arr[k_cell][j_cell-1][i_cell].y + leta_arr[k_cell][j_cell][i_cell].y); //* aj_center;
1799 mat[1][2] = 0.5 * (leta_arr[k_cell][j_cell-1][i_cell].z + leta_arr[k_cell][j_cell][i_cell].z); //* aj_center;
1800
1801 mat[2][0] = 0.5 * (lzet_arr[k_cell-1][j_cell][i_cell].x + lzet_arr[k_cell][j_cell][i_cell].x); //* aj_center;
1802 mat[2][1] = 0.5 * (lzet_arr[k_cell-1][j_cell][i_cell].y + lzet_arr[k_cell][j_cell][i_cell].y); //* aj_center;
1803 mat[2][2] = 0.5 * (lzet_arr[k_cell-1][j_cell][i_cell].z + lzet_arr[k_cell][j_cell][i_cell].z); //* aj_center;
1804
1805 // Contravariant velocity vector `q` at the cell center,
1806 // estimated by averaging face-based contravariant velocities.
1807 PetscReal q[3];
1808 q[0] = 0.5 * (lucont_arr[k_cell][j_cell][i_cell-1].x + lucont_arr[k_cell][j_cell][i_cell].x); // U at cell center
1809 q[1] = 0.5 * (lucont_arr[k_cell][j_cell-1][i_cell].y + lucont_arr[k_cell][j_cell][i_cell].y); // U at cell center
1810 q[2] = 0.5 * (lucont_arr[k_cell-1][j_cell][i_cell].z + lucont_arr[k_cell][j_cell][i_cell].z); // U at cell center
1811
1812 // Solve the 3x3 system `mat * ucat = q` using Cramer's rule.
1813 PetscReal det = mat[0][0] * (mat[1][1] * mat[2][2] - mat[1][2] * mat[2][1]) -
1814 mat[0][1] * (mat[1][0] * mat[2][2] - mat[1][2] * mat[2][0]) +
1815 mat[0][2] * (mat[1][0] * mat[2][1] - mat[1][1] * mat[2][0]);
1816
1817 if (PetscAbsReal(det) < 1.0e-18) {
1818 SETERRQ(PETSC_COMM_SELF, PETSC_ERR_FLOP_COUNT, "Transformation matrix determinant is near zero at cell (%d,%d,%d) \n", i_cell, j_cell, k_cell);
1819 }
1820
1821 PetscReal det_inv = 1.0 / det;
1822
1823 PetscReal det0 = q[0] * (mat[1][1] * mat[2][2] - mat[1][2] * mat[2][1]) -
1824 q[1] * (mat[0][1] * mat[2][2] - mat[0][2] * mat[2][1]) +
1825 q[2] * (mat[0][1] * mat[1][2] - mat[0][2] * mat[1][1]);
1826
1827 PetscReal det1 = -q[0] * (mat[1][0] * mat[2][2] - mat[1][2] * mat[2][0]) +
1828 q[1] * (mat[0][0] * mat[2][2] - mat[0][2] * mat[2][0]) -
1829 q[2] * (mat[0][0] * mat[1][2] - mat[0][2] * mat[1][0]);
1830
1831 PetscReal det2 = q[0] * (mat[1][0] * mat[2][1] - mat[1][1] * mat[2][0]) -
1832 q[1] * (mat[0][0] * mat[2][1] - mat[0][1] * mat[2][0]) +
1833 q[2] * (mat[0][0] * mat[1][1] - mat[0][1] * mat[1][0]);
1834
1835 // Store computed Cartesian velocity in the GLOBAL Ucat array at the
1836 // array index corresponding to the cell's origin node.
1837 gucat_arr[k_cell][j_cell][i_cell].x = det0 * det_inv;
1838 gucat_arr[k_cell][j_cell][i_cell].y = det1 * det_inv;
1839 gucat_arr[k_cell][j_cell][i_cell].z = det2 * det_inv;
1840 }
1841 }
1842 }
1843
1844 // --- 6. Restore Array Access ---
1845 ierr = DMDAVecRestoreArrayRead(user->fda, user->lUcont, &lucont_arr); CHKERRQ(ierr);
1846 ierr = DMDAVecRestoreArrayRead(user->fda, user->lCsi, &lcsi_arr); CHKERRQ(ierr);
1847 ierr = DMDAVecRestoreArrayRead(user->fda, user->lEta, &leta_arr); CHKERRQ(ierr);
1848 ierr = DMDAVecRestoreArrayRead(user->fda, user->lZet, &lzet_arr); CHKERRQ(ierr);
1849 ierr = DMDAVecRestoreArrayRead(user->da, user->lNvert, &lnvert_arr); CHKERRQ(ierr);
1850 ierr = DMDAVecRestoreArrayRead(user->da, user->lAj, &laj_arr); CHKERRQ(ierr);
1851 ierr = DMDAVecRestoreArray(user->fda, user->Ucat, &gucat_arr); CHKERRQ(ierr);
1852
1853 LOG_ALLOW(GLOBAL, LOG_INFO, "Completed Contravariant-to-Cartesian velocity transformation. \n");
1854 PetscFunctionReturn(0);
1855}

1944/**
1945 * @brief Performs a binary search for a key in a sorted array of PetscInt64.
1946 *
1947 * This is a standard binary search algorithm implemented as a PETSc-style helper function.
1948 * It efficiently determines if a given `key` exists within a `sorted` array.
1949 *
1950 * @param[in] n The number of elements in the array.
1951 * @param[in] arr A pointer to the sorted array of PetscInt64 values to be searched.
1952 * @param[in] key The PetscInt64 value to search for.
1953 * @param[out] found A pointer to a PetscBool that will be set to PETSC_TRUE if the key
1954 * is found, and PETSC_FALSE otherwise.
1955 *
1956 * @return PetscErrorCode 0 on success, or a non-zero PETSc error code on failure.
1957 *
1958 * @note The input array `arr` **must** be sorted in ascending order for the algorithm
1959 * to work correctly.
1960 */
1961PetscErrorCode BinarySearchInt64(PetscInt n, const PetscInt64 arr[], PetscInt64 key, PetscBool *found)
1962{
1963 PetscInt low = 0, high = n - 1;
1964
1965 PetscFunctionBeginUser;
1966
1967 // --- 1. Input Validation ---
1968 if (!found) {
1969 SETERRQ(PETSC_COMM_SELF, PETSC_ERR_ARG_NULL, "Output pointer 'found' is NULL in PetscBinarySearchInt64.");
1970 }
1971 if (n > 0 && !arr) {
1972 SETERRQ(PETSC_COMM_SELF, PETSC_ERR_ARG_NULL, "Input array 'arr' is NULL for n > 0.");
1973 }
1974
1975 // Initialize output
1976 *found = PETSC_FALSE;
1977
1978 // --- 2. Binary Search Algorithm ---
1979 while (low <= high) {
1980 // Use this form to prevent potential integer overflow on very large arrays
1981 PetscInt mid = low + (high - low) / 2;
1982
1983 if (arr[mid] == key) {
1984 *found = PETSC_TRUE; // Key found!
1985 break; // Exit the loop
1986 }
1987
1988 if (arr[mid] < key) {
1989 low = mid + 1; // Search in the right half
1990 } else {
1991 high = mid - 1; // Search in the left half
1992 }
1993 }
1994
1995 PetscFunctionReturn(0);
1996}
