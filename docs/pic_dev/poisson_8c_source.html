5static void Calculate_Covariant_metrics(double g[3][3], double G[3][3])
6{
7 /*
8 | csi.x csi.y csi.z |-1 | x.csi x.eta x.zet |
9 | eta.x eta.y eta.z | = | y.csi y.eta y.zet |
10 | zet.x zet.y zet.z | | z.csi z.eta z.zet |
11
12 */
13 const double a11=g[0][0], a12=g[0][1], a13=g[0][2];
14 const double a21=g[1][0], a22=g[1][1], a23=g[1][2];
15 const double a31=g[2][0], a32=g[2][1], a33=g[2][2];
16
17 double det= a11*(a33*a22-a32*a23)-a21*(a33*a12-a32*a13)+a31*(a23*a12-a22*a13);
18
19 G[0][0] = (a33*a22-a32*a23)/det, G[0][1] = - (a33*a12-a32*a13)/det, G[0][2] = (a23*a12-a22*a13)/det;
20 G[1][0] = -(a33*a21-a31*a23)/det, G[1][1] = (a33*a11-a31*a13)/det, G[1][2] = - (a23*a11-a21*a13)/det;
21 G[2][0] = (a32*a21-a31*a22)/det, G[2][1] = - (a32*a11-a31*a12)/det, G[2][2] = (a22*a11-a21*a12)/det;
22};

24static void Calculate_normal(Cmpnts csi, Cmpnts eta, Cmpnts zet, double ni[3], double nj[3], double nk[3])
25{
26 double g[3][3];
27 double G[3][3];
28
29 g[0][0]=csi.x, g[0][1]=csi.y, g[0][2]=csi.z;
30 g[1][0]=eta.x, g[1][1]=eta.y, g[1][2]=eta.z;
31 g[2][0]=zet.x, g[2][1]=zet.y, g[2][2]=zet.z;
32
33 Calculate_Covariant_metrics(g, G);
34 double xcsi=G[0][0], ycsi=G[1][0], zcsi=G[2][0];
35 double xeta=G[0][1], yeta=G[1][1], zeta=G[2][1];
36 double xzet=G[0][2], yzet=G[1][2], zzet=G[2][2];
37
38 double nx_i = xcsi, ny_i = ycsi, nz_i = zcsi;
39 double nx_j = xeta, ny_j = yeta, nz_j = zeta;
40 double nx_k = xzet, ny_k = yzet, nz_k = zzet;
41
42 double sum_i=sqrt(nx_i*nx_i+ny_i*ny_i+nz_i*nz_i);
43 double sum_j=sqrt(nx_j*nx_j+ny_j*ny_j+nz_j*nz_j);
44 double sum_k=sqrt(nx_k*nx_k+ny_k*ny_k+nz_k*nz_k);
45
46// *Ai = sqrt( g[0][0]*g[0][0] + g[0][1]*g[0][1] + g[0][2]*g[0][2] ); // area
47// *Aj = sqrt( g[1][0]*g[1][0] + g[1][1]*g[1][1] + g[1][2]*g[1][2] );
48// *Ak =sqrt( g[2][0]*g[2][0] + g[2][1]*g[2][1] + g[2][2]*g[2][2] );
49
50 nx_i /= sum_i, ny_i /= sum_i, nz_i /= sum_i;
51 nx_j /= sum_j, ny_j /= sum_j, nz_j /= sum_j;
52 nx_k /= sum_k, ny_k /= sum_k, nz_k /= sum_k;
53
54 ni[0] = nx_i, ni[1] = ny_i, ni[2] = nz_i;
55 nj[0] = nx_j, nj[1] = ny_j, nj[2] = nz_j;
56 nk[0] = nx_k, nk[1] = ny_k, nk[2] = nz_k;
57}

949/**
950 * @brief Performs the projection step to enforce an incompressible velocity field.
951 *
952 * @details
953 * This function executes the final "correction" stage of a fractional-step (projection)
954 * method for solving the incompressible Navier-Stokes equations. After solving the
955 * Pressure Poisson Equation to get a pressure correction `Phi`, this function uses
956 * `Phi` to correct the intermediate velocity field, making it divergence-free.
957 *
958 * The main steps are:
959 * 1. **Calculate Pressure Gradient**: At each velocity point (on the cell faces), it
960 * computes the gradient of the pressure correction field (``). This is done using
961 * finite differences on a generalized curvilinear grid.
962 *
963 * 2. **Correct Velocity Field**: It updates the contravariant velocity components `Ucont`
964 * by subtracting the scaled pressure gradient. The update rule is:
965 * `U_new = U_intermediate - t * G * `, where `G` is the metric tensor `g_ij`
966 * that transforms the gradient into contravariant components.
967 *
968 * 3. **Boundary-Aware Gradients**: The gradient calculation is highly sensitive to
969 * boundaries. The code uses complex, dynamic stencils that switch from centered
970 * differences to one-sided differences if a standard stencil would use a point
971 * inside an immersed solid (`nvert > 0.1`). This preserves accuracy at the
972 * fluid-solid interface.
973 *
974 * 4. **Periodic Boundary Updates**: Includes dedicated loops to correctly calculate
975 * the pressure gradient at the domain edges for periodic boundary conditions.
976 *
977 * 5. **Finalization**: After correcting `Ucont`, it calls helper functions to convert
978 * the velocity back to Cartesian coordinates (`Contra2Cart`) and update all
979 * ghost cell values.
980 *
981 * @param[in, out] user Pointer to the UserCtx struct, containing simulation context,
982 * grid metrics, and the velocity/pressure fields.
983 *
984 * @return PetscErrorCode 0 on success, or a non-zero error code on failure.

2188/**
2189 * @brief Assembles the Left-Hand Side (LHS) matrix for the Pressure Poisson Equation.
2190 *
2191 * @details
2192 * This function constructs the sparse matrix `A` (the LHS) for the linear system `Ax = B`,
2193 * which is the Pressure Poisson Equation (PPE). The matrix `A` represents a discrete
2194 * version of the negative Laplacian operator (-), tailored for a general curvilinear,
2195 * staggered grid.
2196 *
2197 * The assembly process is highly complex and follows these main steps:
2198 *
2199 * 1. **Matrix Initialization**: On the first call, it allocates a sparse PETSc matrix `A`
2200 * pre-allocating space for a 19-point stencil per row. On subsequent calls, it
2201 * simply zeroes out the existing matrix.
2202 *
2203 * 2. **Metric Tensor Calculation**: It first computes the 9 components of the metric
2204 * tensor (`g11`, `g12`, ..., `g33`) at the cell faces. These `g_ij` coefficients
2205 * are essential for defining the Laplacian operator in generalized curvilinear
2206 * coordinates and account for grid stretching and non-orthogonality.
2207 *
2208 * 3. **Stencil Assembly Loop**: The function iterates through every grid point `(i,j,k)`.
2209 * For each point, it determines the matrix row entries based on its status:
2210 * - **Boundary/Solid Point**: If the point is on a domain boundary or inside an
2211 * immersed solid (`nvert > 0.1`), it sets an identity row (`A(row,row) = 1`),
2212 * effectively removing it from the linear solve.
2213 * - **Fluid Point**: For a fluid point, it computes the 19 coefficients of the
2214 * finite volume stencil. This involves summing contributions from each of the
2215 * six faces of the control volume around the point.
2216 *
2217 * 4. **Boundary-Aware Stencils**: The stencil calculation is critically dependent on the
2218 * state of neighboring cells. The code contains intricate logic to check if neighbors
2219 * are fluid or solid. If a standard centered-difference stencil would cross into a
2220 * solid, the scheme is automatically adapted to a one-sided difference to maintain
2221 * accuracy at the fluid-solid interface.
2222 *
2223 * 5. **Matrix Finalization**: After all rows corresponding to the local processor's
2224 * grid points have been set, `MatAssemblyBegin/End` is called to finalize the
2225 * global matrix, making it ready for use by a linear solver.
2226 *
2227 * @param[in, out] user Pointer to the UserCtx struct, containing all simulation context,
2228 * grid data, and the matrix `A`.
2229 *
2230 * @return PetscErrorCode 0 on success, or a non-zero error code on failure.
2231 */

4170 ierr = KSPSetUp(subksp); CHKERRQ(ierr);

4185 ierr = KSPSetFromOptions(mgksp); CHKERRQ(ierr);
4186 ierr = KSPSetUp(mgksp); CHKERRQ(ierr);

4191 user = mgctx[l].user;
4192 MatNullSpaceDestroy(&user[bi].nullsp);
4193 MatDestroy(&user[bi].A);
4194 user[bi].assignedA = PETSC_FALSE;

4196 MatDestroy(&user[bi].MR);
4197 MatDestroy(&user[bi].MP);
4198 } else if (l==0 && immersed) {

4200 }

4202 VecDestroy(&user[bi].R);
4203 }

4207 VecDestroy(&mgctx[usermg->mglevels-1].user[bi].B);
